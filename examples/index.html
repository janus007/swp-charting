Komplet Konfigurationsreference

// createChart(container, options)

{
  // === CHART DIMENSIONER ===
  width: 650,                    // number (default: container width)
  height: 350,                   // number (default: 350)
  padding: {
    top: 30,                     // number (default: 30)
    right: 40,                   // number (default: 40)
    bottom: 40,                  // number (default: 40)
    left: 60,                    // number (default: 60)
  },

  // === X-AKSE ===
  xAxis: {
    categories: ['Jan', 'Feb', ...]  // string[] - påkrævet
  },

  // === Y-AKSE ===
  yAxis: {
    min: 0,                      // number (default: 0)
    max: 1000,                   // number (default: auto)
    ticks: 5,                    // number - antal grid linjer
    format: (v) => `${v} kr`,   // (value: number) => string
  },

  // === SERIER ===
  series: [{
    name: 'Serie navn',          // string - påkrævet
    color: '#8b5cf6',            // string - påkrævet
    data: [{ x: 'Jan', y: 100 }], // DataPoint[] - påkrævet
    type: 'line',                // 'line' | 'bar' (default: 'line')
    showArea: true,              // boolean (default: true)

    // LINE STYLING (kun for type: 'line')
    line: {
      width: 2,                  // number (default: 2)
      opacity: 1,                // number 0-1 (default: 1)
      curve: 'smooth',           // 'smooth' | 'linear' (default: 'smooth')
    },

    // POINT STYLING (kun for type: 'line')
    point: {
      style: 'circle',           // 'circle' | 'dot' (default: 'circle')
      radius: 6,                 // number (default: 6 for circle, 4 for dot)
      fill: '#fff',              // string (default: '#fff' for circle)
      stroke: '#8b5cf6',         // string (default: series color)
      strokeWidth: 2,            // number (default: 2)
      hoverRadius: 8,            // number (default: radius + 2)
    },

    // AREA STYLING (kun for type: 'line')
    area: {
      opacity: 0.3,              // number 0-1 (default: 0.3)
      gradient: {                // GradientConfig | boolean (default: true)
        startOpacity: 0.4,       // number (default: 0.4)
        endOpacity: 0.05,        // number (default: 0.05)
      },
    },

    // BAR STYLING (kun for type: 'bar')
    bar: {
      width: 20,                 // number | 'auto' (default: 20)
      radius: 0,                 // number (default: 0)
      opacity: 1,                // number 0-1 (default: 1)
    },
  }],

  // === TOOLTIP ===
  tooltip: {                       // TooltipConfig | boolean (default: true)
    animation: {
      duration: 150,             // number ms (default: 150)
      easing: 'ease-out',        // string (default: 'ease-out')
    },
    render: (data) => `...`,      // custom render function
  },

  // === LEGEND ===
  legend: {                        // LegendConfig | boolean (default: false)
    position: 'top',              // 'top' | 'bottom' | 'left' | 'right'
    align: 'center',             // 'start' | 'center' | 'end'
  },

  // === ANIMATION ===
  animation: {                     // AnimationConfig | boolean (default: true)
    enabled: true,               // boolean (default: true)
    duration: 600,               // number ms (default: 600)
    easing: 'ease-out',          // string (default: 'ease-out')
  },
}

// === CHART INSTANCE METODER ===
const chart = createChart(container, options);
chart.update({ series: [...] });  // Opdater chart
chart.resize();                   // Gentegn ved resize
chart.destroy();                  // Fjern og ryd op
